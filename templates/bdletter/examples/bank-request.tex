\documentclass{../bdletter}

\LetterSetup{

  recipient    = {জনাব শাখা ব্যবস্থাপক},
  organization = {সোনালী ব্যাংক লিমিটেড},
  address      = {মতিঝিল শাখা, ঢাকা-১০০০},
  subject      = {ব্যাংক হিসাবের বিবরণী প্রদান প্রসঙ্গে},
  salutation   = {মহোদয়},
  closing      = {বিনীত নিবেদক},
  name         = {মোসাম্মৎ সালমা বেগম},
  position     = {হিসাব নং: ১২৩৪৫৬৭৮৯০},
  sender_org   = {বাড়ি নং: ১২, রোড নং: ৫, ধানমন্ডি, ঢাকা-১২০৯},
  contact      = {মোবাইল: ০১৯১২-০০০০০০},
}

\begin{document}
\PrintLetterHeader

বিনীত নিবেদন এই যে, আমি আপনার শাখায় একজন নিয়মিত আমানতকারী। আমার হিসাব
নম্বর ১২৩৪৫৬৭৮৯০। আমার ব্যক্তিগত প্রয়োজনে গত ০১/০৭/২০২৩ খ্রিঃ তারিখ হতে
৩১/১২/২০২৩ খ্রিঃ তারিখ পর্যন্ত সময়ের ব্যাংক হিসাবের বিবরণী (Bank Statement)
প্রয়োজন।

অতএব, মহোদয়ের নিকট বিনীত প্রার্থনা, উপরোক্ত সময়ের ব্যাংক হিসাবের বিবরণী
প্রদান করে বাধিত করবেন।

\PrintLetterFooter
\end{document}
